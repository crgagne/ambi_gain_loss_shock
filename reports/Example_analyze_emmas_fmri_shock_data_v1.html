# convert data into a form that is useable for my model scripts. 

folder = '../output/output_emmas/tasklogs/'
    
MIDs = []
for sub in range(32):
    trial_array = np.empty((200,8))
    
    # basic 
    trial_array[:,0] = dictt['choices'][:,sub] # check this is right/left 
    trial_array[:,1] =dictt['ProportionNoughts_L'][sub,:]
    trial_array[:,2] =dictt['ProportionNoughts_R'][sub,:]
    trial_array[:,3] =dictt['Magnitude_Left'][sub,:]/100.0
    trial_array[:,4] =dictt['Magnitude_Right'][sub,:]/100.0
    trial_array[:,5] =dictt['Ambiguity'][sub,:]
    trial_array[:,6] =dictt['no_noughtsrevealedTOSAVEtemp'][:,sub]
    trial_array[:,7] =dictt['no_tokensrevealedTOSAVEtemp'][:,sub]
    
    # make dataFrame
    trial_data = pd.DataFrame(data=trial_array,columns=['resp_r_1','prob_o_l','prob_o_r','mag_left','mag_right','info_amb','revealed_o_r','num_revealed'])
    
    # ambiguous trials (right=ambiguous)
    trial_data['info_amb_sqrt']=np.sqrt(trial_data['info_amb'])
    trial_data.loc[trial_data['resp_r_1']>1,'resp_r_1']=np.nan
    trial_data['resp_amb_1'] = trial_data['resp_r_1']
    trial_data['ambig_r']=(trial_data['info_amb_sqrt']!=1).astype('int')
    trial_data['ambig_l']=np.zeros(len(trial_data))
    trial_data['gain_or_loss_trial']='shock'
    trial_data['mag_ambig']=trial_data['mag_right']
    trial_data['mag_unambig']=trial_data['mag_left']
    trial_data['prob_o_unambig']=trial_data['prob_o_l']
    trial_data['prob_o_ambig']=trial_data['prob_o_r']
    trial_data['revealed_o_ambig']=trial_data['revealed_o_r'] # right side is ambig
    trial_data['revealed_x_ambig']=trial_data['num_revealed']-trial_data['revealed_o_ambig']
    trial_data['prob_o_ambig_bayes']=(trial_data['revealed_o_r']+1)/(trial_data['num_revealed']+2)
    trial_data['MID']='S'+str(sub)
    MIDs.append('S'+str(sub))
    
    if not os.path.isdir(folder):
        os.mkdir(folder)
    trial_data.to_csv(folder+'/S'+str(sub)+'.csv')

def plot_params(df,stripplot=False,outlier_cutoff=None):
    plt.style.use(['seaborn-white', 'seaborn-paper'])
    matplotlib.rc("font", family="Times New Roman")
    sns.set_context('talk')
    sns.set_style('white',{'figure.facecolor':'white'})
    
    
    if outlier_cutoff is not None:
        df = df[(df.beta>-1.0*outlier_cutoff)&(df.beta<outlier_cutoff)]
    
    axis = sns.barplot(x='parameter',y='beta',data=df,color=[.4,.4,.4],ci=95,alpha=0.4)
    
    if stripplot:
        sns.stripplot(x="parameter", y="beta", data=df,color=[.4,.4,.4],alpha=0.2,jitter=True);

        
    current_palette=sns.color_palette()
    fig = plt.gcf()
    fig.suptitle('Model Parameters: '+modelname+' '+task,fontsize=12,x=0.55)
    sns.despine(ax=axis)
    axis.set_ylabel('beta (Prob Choose Right (except on Ambig))',fontsize=12)
    axis.set_xlabel('parameter',fontsize=12)
    axis.set_xticklabels(df.parameter.unique(),rotation=45,fontsize=12,ha='right')
    axis = plt.gca()
    
    # change name if needed
    xlabels = axis.get_xticklabels()

    fig.suptitle('')
    axis.set_title('Model Parameters (Across all Subjects)')
    axis.set_xlabel('Parameter')
    axis.set_ylabel('Group Regression Coefficients \n (Probability Choosing Ambig)')
    plt.tight_layout()
    return(fig)

params = df.parameter.unique() 

sns.set_context('paper')

traits = ['STAI_Trait']
for t,trait in enumerate(traits):
    fig,axes = plt.subplots(1,len(df['parameter'].unique()),figsize=(len(df['parameter'].unique())*2,3))
    axes = axes.flatten()
    i=0
    for param in params:
        
        dft = df.loc[df['parameter']==param,]
        x = dft[trait].as_matrix()
        y = dft['beta'].as_matrix()
        nans = np.logical_and(~np.isnan(x), ~np.isnan(y))
        x=x[nans]
        y=y[nans]
        (r,p) =scipy.stats.pearsonr(x,y)
        r = np.round(r,2)
        p = np.round(p,3)
        axes[i].scatter(x,y,alpha=0.5)
        
        X = sm.add_constant(x, prepend=False)
        res_ols = sm.OLS(y, X).fit()
        p_ols=res_ols.pvalues[0]
        b_ols=res_ols.params[0]
        intercept = res_ols.params[1]
        axes[i].plot(X[:,0],b_ols*X[:,0]+intercept,color='b')
        
        
        axes[i].set_title(param+'\n r='+str(r)+'p='+str(p))
        axes[i].set_xlabel(trait)
        sns.despine()
        print(param)
        i+=1
